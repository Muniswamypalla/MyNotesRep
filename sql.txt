create table APP_USER (id NUMBER(10) NOT NULL, sso_id VARCHAR2(30) NOT NULL, password VARCHAR2(100) NOT NULL, 
first_name VARCHAR2(30) NOT NULL, last_name  VARCHAR2(30) NOT NULL, email VARCHAR2(30) NOT NULL, PRIMARY KEY (id), 
UNIQUE (sso_id));



create table USER_PROFILE(
   id NUMBER(10) NOT NULL,
   type VARCHAR2(30) NOT NULL,
   PRIMARY KEY (id),
   UNIQUE (type)
);


CREATE TABLE APP_USER_USER_PROFILE (
    user_id NUMBER(10) NOT NULL,
    user_profile_id NUMBER(10) NOT NULL,
    PRIMARY KEY (user_id, user_profile_id),
    CONSTRAINT FK_APP_USER FOREIGN KEY (user_id) REFERENCES APP_USER (id),
    CONSTRAINT FK_USER_PROFILE FOREIGN KEY (user_profile_id) REFERENCES USER_PROFILE (id)
);
